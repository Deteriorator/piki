import java.awt.EventQueue;
import java.awt.GridLayout;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.io.BufferedReader;
import java.io.FileNotFoundException;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;

import javax.swing.JFileChooser;
import javax.swing.JFrame;
import javax.swing.JMenu;
import javax.swing.JMenuBar;
import javax.swing.JMenuItem;
import javax.swing.JPanel;
import javax.swing.JScrollPane;
import javax.swing.JTextArea;
import javax.swing.filechooser.FileNameExtensionFilter;

/**
 * 2013年6月3日下午，北大java上机考试 要求：写一个文本编辑器，具有新建，打开，保存，编辑(复制，剪切，粘贴)功能。

public class MySimpleEditor {

public static void main(String[] args) {
EventQueue.invokeLater(new Runnable() {
@Override
public void run() {
new EditorFrame().setVisible(true);
}
});
}

}

class EditorFrame extends JFrame implements ActionListener {
private int DEFAULT_WIDTH = 800;
private int DEFAULT_HEIGH = 500;
private JMenuBar menubar;
private JTextArea editor;

public EditorFrame() {
this.setSize(DEFAULT_WIDTH, DEFAULT_HEIGH);
this.setTitle("MySimpleEditor");
this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

initMenuBar();
initEditor();
}

private void initMenuBar() {
menubar = new JMenuBar();
this.setJMenuBar(menubar);

JMenu file = new JMenu("File");
JMenu edit = new JMenu("Edit");
menubar.add(file);
menubar.add(edit);

addMenu(file, "create");
addMenu(file, "open");
addMenu(file, "save");

addMenu(edit, "copy");
addMenu(edit, "cut");
addMenu(edit, "paste");
}

private void addMenu(JMenu menu, String itemName) {
JMenuItem item = new JMenuItem(itemName);
item.addActionListener(this);
menu.add(item);
}

private void initEditor() {
JPanel panel = new JPanel();
panel.setLayout(new GridLayout(1, 1));

editor = new JTextArea();
JScrollPane scroll = new JScrollPane(editor);
panel.add(scroll);
this.getContentPane().add(panel);
}

@Override
public void actionPerformed(ActionEvent e) {
System.out.println(e.getActionCommand());
JFileChooser chooser = new JFileChooser();
FileNameExtensionFilter filter = new FileNameExtensionFilter(
"txt Txt file", "txt");
chooser.setFileFilter(filter);

switch (e.getActionCommand()) {
case "create":
editor.setText("");
break;

case "open":
editor.setText("");
int returnVal = chooser.showOpenDialog(getParent());
if (returnVal == JFileChooser.APPROVE_OPTION) {
try {
BufferedReader reader = new BufferedReader(new FileReader(
chooser.getSelectedFile()));
String line;
while ((line = reader.readLine()) != null) {
editor.append(line);
editor.append("\n");
}
reader.close();
} catch (FileNotFoundException e1) {
// TODO Auto-generated catch block
e1.printStackTrace();
} catch (IOException e1) {
// TODO Auto-generated catch block
e1.printStackTrace();
}
}

break;
case "save":
chooser.showSaveDialog(getParent());
try {
FileWriter writer = new FileWriter(chooser.getSelectedFile());
writer.write(editor.getText());
writer.close();
} catch (IOException e1) {
// TODO Auto-generated catch block
e1.printStackTrace();
}
break;

case "copy":
editor.copy();
break;

case "cut":
editor.cut();
break;

case "paste":
editor.paste();
break;
}
}
}
